\subsection*{$100(1-\alpha)$\si{\percent}-BI voor de binomiale succeskans $p$ (Clopper-Pearson methode)}

    \begin{enumerate}
        \item Bereken ondergrens van het ineterval: de kleinste succeskans $p_1$ zodat geldt $P(X \geq k) \geq \alpha/2$.
        \item Bereken bovengrens van het interval: de grootste succeskans $p_2$ zodat geldt $P(X \leq k) \geq \alpha/2$.
    \end{enumerate}
    \begin{multicols}{2}

            \begin{enumerate}
                \item Gebruik numerieke solver met 
                \begin{itemize}
                    \item[$E_1$] $= 1 - \binomcdf(n, p=X, k-1)$
                    \item[$E_2$] $=\alpha/2$
                \end{itemize}
                \item Gebruik numerieke solver met
                \begin{itemize}
                    \item[$E_1$] $= \binomcdf(n, p=X, k)$
                    \item[$E_2$] $=\alpha/2$
                \end{itemize}
            \end{enumerate}

            \columnbreak

            \centering\textbf{Casio fx-CG50}
            \begin{enumerate}
                \item Gebruik numerieke solver met \[\solveNC\left(1-\binomcdfC(k-1,n,x)=\alpha/2, x\right)\]
                \item Gebruik numerieke solver met \[\solveNC\left(\binomcdfC(k,n,x)=\alpha/2, x\right)\]
            \end{enumerate}
    \end{multicols}

\subsection*{Stappenplan voor iedere hypothesetoets}
\begin{enumerate}
    \item Definieer de nulhypothese $H_0$ en de alternatieve hypothese $H_1$.
    \item Bepaal het significantieniveau $\alpha$ (kans op onterecht verwerpen van $H_0$ $\rightarrow$ type-I fout).
    \item Verzamel data voor de toetsingsgrootheid.
    \item Bereken de toetsingsgrootheid.
    \item Bepaal de toetsuitslag (wel of niet verwerpen van $H_0$).
    \item Vertaal de toetsuitslag naar de originele probleemcontext.
\end{enumerate}

\newpage 

\subsection*{Toetsuitslag bepalen}
\begin{itemize}
    \item De toetsingsgrootheid volgt, uitgaande van $H_0$, een bepaalde kansverdeling.
    \item Op basis van deze kansverdeling kun je (i) een kritiek gebied bepalen, of (ii) een $p$-waarde berekenen.
    \begin{itemize}
        \item \textbf{Kritiek gebied:} verwerp $H_0$ alleen als de geobserveerde toetsingsgrootheid in het kritieke gebied ligt.
        \item \textbf{$p$-waarde:} verwerp $H_0$ alleen als $p \le \alpha$ (links-/rechtszijdig) of $p \le \frac{\alpha}{2}$ (tweezijdig).
    \end{itemize}
\end{itemize}
