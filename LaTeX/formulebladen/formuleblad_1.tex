Gegeven een steekproef met $n$ waarnemingen $x_1, x_2, \ldots, x_n$:

\textbf{Steekproefgemiddelde}
\[
    \overline{x} = \frac{\sum_{i} x_i}{n} = \frac{x_1 + x_2 + \ldots + x_n}{n}
\]

\textbf{Steekproefvariantie en steekproefstandaardafwijking:}
\begin{align*}
    s^2 &= \frac{\sum_{i=1}^{n} (x_i-\overline{x})^2}{n-1} = \frac{(x_1-\overline{x})^2 + (x_2-\overline{x})^2 + \ldots + (x_n - \overline{x})^2}{n-1} \\[1ex]
    s   &= \sqrt{s^2} = \sqrt{ \frac{(x_1-\overline{x})^2 + (x_2-\overline{x})^2 + \ldots + (x_n - \overline{x})^2}{n-1} }
\end{align*}

\textbf{Rekenregels kansrekening ($\cup$ betekent \enquote{of}, $\cap$ betekent \enquote{en}, $\neg$ betekent \enquote{niet}, $\vert$ betekent \enquote{gegeven}):}
{
    \addtolength{\jot}{1em}
    \begin{align*}
        P(A \cup B)  &= P(A) + P(B) - P(A \cap B) \tag{optelregel} \\
        P(B)                &= 1 - P(\neg B) \tag{complementregel} \\
        P(A \mid B)         &= \frac{P(A \cap B)}{P(B)} \tag{conditionele kansen}
    \end{align*}
}

\textbf{Speciale kansverdelingen:}
\begin{itemize}
    \item \textbf{Binomiale verdeling}: aantal successen tellen bij onafhankelijke (Bernoulli-)kansexperimenten met twee mogelijke uitkomsten: succes / mislukking.
    \begin{itemize}
        \item {\itshape Parameters:} aantal Bernoulli-experimenten $n$, succeskans per experiment $p$
    \end{itemize}
    
    \item \textbf{Poissonverdeling:} aantal \enquote{gebeurtenissen} tellen in een \enquote{meetinterval} van tijd of ruimte.
    \begin{itemize}
        \item {\itshape Parameters:} gemiddelde aantal gebeurtenissen $\lambda$ per meeteenheid (tijd / ruimte), het aantal meeteenheden $t$ (bijv. $t = 7$ als een week lang wordt gemeten, en de meeteenheid is \enquote{dag}).
    \end{itemize}

    \item \textbf{Exponentiële verdeling:} de tijd / afstand tot de volgende \enquote{gebeurtenis}.
    \begin{itemize}
        \item {\itshape Parameter:} gemiddelde aantal gebeurtenissen $\lambda$ per meeteenheid (tijd / ruimte).  
    \end{itemize}
\end{itemize}

\textbf{$z$-score:}
\[
    z = \frac{x - \mu}{\sigma}\\
\]

\textbf{Centrale limietstelling:}

Gegeven zijn $n$ kansvariabelen $X_1, X_2, \ldots, X_n$, die onderling onafhankelijk zijn en dezelfde kansverdeling volgen 
met verwachtingswaarde $\mu$ en standaardafwijking $\sigma$.
Er geldt (bij benadering) het volgende:
\begin{itemize}
    \item de som $\sum_{i=1}^{n} X_i = X_1 + X_2 + \ldots + X_n$ is normaal verdeeld met parameters $n \cdot \mu$ en $\sqrt{n} \cdot \sigma$.
    \item het gemiddelde $\overline{X} = \frac{X_1 + X_2 + \ldots + X_n}{n}$ is normaal verdeeld met parameters $\mu$ en $\frac{\sigma}{\sqrt{n}}$.
\end{itemize}
