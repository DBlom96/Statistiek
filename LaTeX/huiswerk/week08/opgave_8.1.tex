\question{8.1}{
    Een machine vult zakken met meel.
    Het gewicht van het meel in een willekeurige zak is te beschouwen als een trekking uit een normale verdeling
    met onbekende $\mu$ en een standaarddeviatie van $100$ gram.
    Om het schattingsinterval te berekenen voor $\mu$ weegt men de inhoud van $25$ zakken meel.
    Het gemiddelde gewicht van het meel per zak bedroeg $20.142$ kg.   
}
\begin{enumerate}[label=(\alph*)]
    \item Bereken een schattingsinterval voor $\mu$ (de verwachtingswaarde van het gewicht van het meel) dat een betrouwbaarheid heeft van \SI{95}{\percent}.
    \answer{
        De kansvariabele $X$ meet het gewicht (in gram) van een willekeurige zak meel.
        Er is gegeven dat $X$ normaal verdeeld is met onbekende $\mu = \text{ ?}$ en standaardafwijking $\sigma = 100$.
        Als we $25$ willekeurige zakken meel gaan wegen, hebben we te maken met $25$ kansvariabelen $X_1, X_2, \ldots, X_{25}$.
        Volgens de centrale limietstelling geldt dat het theoretische steekproefgemiddelde $\samplemeanvarX$ van $25$ zakken meel normaal verdeeld is met
        verwachtingswaarde $\mu = \text{ ?}$ en standaardafwijking $\frac{\sigma}{\sqrt{n}} = \frac{100}{\sqrt{25}} = 20$.

        Gegeven is een geobserveerde steekproef $x_1, x_2, \ldots, x_{25}$ met steekproefgemiddelde $\samplemeanx = 20142$ gram. 
        Aangezien we een betrouwbaarheid van \SI{95}{\percent} willen, is $\alpha = 0.05$ en 
        \[
            z_{\alpha/2} = \invnorm(\opp=1-\alpha/2,\mu=0,\sigma=1) \approx 1.9600.
        \]
        Het \SI{95}{\percent}-betrouwbaarheidsinterval voor $\mu$ op basis van deze geobserveerde steekproef is dan gelijk aan 
        \begin{align*}
            &[\samplemeanx - z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}, \samplemeanx + z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}] \\
            &=[20142 - 1.9600 \cdot \frac{100}{\sqrt{25}}, 20142 + 1.9600 \cdot \frac{100}{\sqrt{25}}] \\
            &=[20102.8, 20181.2]
        \end{align*}
        Met \SI{95}{\percent} betrouwbaarheid ligt het gemiddelde gewicht van een zak meel tussen de $20102.8$ en $20181.2$ gram.
    }

    \item Bereken een schattingsinterval voor $\mu$ dat een betrouwbaarheid heeft van \SI{99}{\percent}.
    \answer{
        Aangezien we in dit geval een betrouwbaarheid van \SI{99}{\percent} willen, geldt nu dat $\alpha = 0.01$ en 
        \[
            z_{\alpha/2} = \invnorm(\opp=1-\alpha/2, \mu=0,\sigma=1) \approx 2.5758.
        \]
        Het \SI{99}{\percent}-betrouwbaarheidsinterval voor $\mu$ is dan gelijk aan 
        \begin{align*}
            &[\samplemeanx - z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}, \samplemeanx + z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}] \\
            &=[20142 - 2.5758 \cdot \frac{100}{\sqrt{25}}, 20142 + 2.5758 \cdot \frac{100}{\sqrt{25}}] \\
            &=[20090.5, 20193.5]
        \end{align*}
        Met \SI{99}{\percent} betrouwbaarheid ligt het gemiddelde gewicht van een zak meel tussen de $20090.5$ en $20193.5$ gram.
    }

    \item Hoe groot moet men de steekproefomvang $n$ kiezen om tot een \SI{95}{\percent}-betrouwbaar\-heidsinterval voor $\mu$ te komen dat een breedte heeft van $20$ eenheden (gram)?
    \answer{
        De breedte van het betrouwbaarheidsinterval
        \[
            [\samplemeanx - z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}, \samplemeanx + z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}}]
        \]
        is gelijk aan $\text{rechtergrens} - \text{linkergrens} = 2z_{\alpha/2} \cdot \frac{\sigma}{\sqrt{n}} = 20$.
        De intervalbreedte is gelijk aan twee keer de maximale afwijking $a$, oftewel $a = 10$.
        Verder geldt bij een onbetrouwbaarheidsniveau $\alpha = 0.05$ dat $z_{\alpha/2} \approx 1.9600$.
        Aangezien we een betrouwbaarheid van \SI{95}{\percent} willen en een maximale afwijking $a = 10$ gram, krijgen we
        \begin{align*}
            n \ge \left( \frac{z_{\alpha/2} \cdot \sigma}{a}\right)^2 = \left(\frac{1.9600 \cdot 100}{10}\right)^2 \approx 384.16
        \end{align*}
        Om de garantie van een intervalbreedte van maximaal $20$ gram te waarborgen, moeten we naar boven afronden, dus de steekproef moet uit minimaal $n=385$ zakken meel bestaan.
    }
\end{enumerate}
